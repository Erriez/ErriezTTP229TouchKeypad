10 ## Library features
11 
12 * 3.3V and 5V
13 * Two wire interface (Not I2C compatible)
14 * Single 4x4 touch keypad
15 * Dual 4x4 touch keypads
16 * Interrupt based
17 * UNO/Mini/Micro/Mega2560/Leonardo, ESP8266, ESP32 and DUE
18 
19 
20 ## Design notes
21 
22 * The PCB must be mounted floated, otherwise key presses don't work reliable.
23 * Connect pins 3 and 4 on connector P1 for 16-pin button mode.
24 * Smaller keypads with less pins may work, but are not tested.

33 ## Getting started
34 
35 ```c++
36 #include <ErriezTTP229.h>
37 
38 // TTP229 pin defines
39 #if defined(ARDUINO_ARCH_AVR)
40 #define TTP229_SDO_PIN 2 // SDO to interrupt pin 2 (INT0) or pin 3 (INT1)
41 #define TTP229_SCL_PIN 3 // SCL to any DIGITAL IO pin
42 #elif defined(ARDUINO_ARCH_ESP8266)
43 #define TTP229_SDO_PIN D1 // Interrupt pin
44 #define TTP229_SCL_PIN D2 // Any pin
45 #elif defined(ARDUINO_ARCH_ESP32)
46 #define TTP229_SDO_PIN 16 // Keep GPIO0 low during programming
47 #define TTP229_SCL_PIN 4
48 #else
49 #error "May work, but not tested on this target"
50 #endif
51 
52 // Create keypad object
53 ErriezTTP229 ttp229;
54 
55 
56 #if defined(ARDUINO_ARCH_ESP8266) || defined(ARDUINO_ARCH_ESP32)
57 ICACHE_RAM_ATTR
58 #endif
59 void keyChange()
60 {
61  // A key press changed
62  ttp229.keyChange = true;
63 }
64 
65 void setup()
66 {
67  // Initialize serial
68  Serial.begin(115200);
69  while (!Serial) {
70  ;
71  }
72  Serial.begin(115200);
73  Serial.println(F("\nErriez TTP229 single 16-keys keypad example"));
74 
75  // Initialize keypad with interrupt
76  ttp229.begin(TTP229_SCL_PIN, TTP229_SDO_PIN);
77 
78  // Initialize interrupt pin on SD0
79  attachInterrupt(digitalPinToInterrupt(TTP229_SDO_PIN), keyChange, FALLING);
80 }
81 
82 void loop()
83 {
84  // Print key
85  if (ttp229.keyChange) {
86  Serial.println(ttp229.GetKey16());
87  ttp229.keyChange = false;
88  }
89 }
90 ```
